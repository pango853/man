


###. Zend Server

#. Installation
	[Zend]
	name=zend-server
	baseurl=http://repos.zend.com/zend-server/6.3/rpm/$basearch
	enabled=1
	gpgcheck=1
	gpgkey=http://repos.zend.com/zend.key
	[Zend_noarch]
	name=zend-server - noarch
	baseurl=http://repos.zend.com/zend-server/6.3/rpm/noarch
	enabled=1
	gpgcheck=1
	gpgkey=http://repos.zend.com/zend.key
Community edition
$ sudo yum install zend-server-ce-php-5.3
Full?
$ sudo yum install zend-server-php-5.4

	! 5.4 is your PHP version

$ sudo /usr/local/zend/bin/zendctl.sh restart

#. Reference
	/usr/local/zend
	/usr/local/zend/etc/sites.d
	/etc/httpd/conf/httpd.conf
	/usr/local/zend/bin/zendctl.sh start|stop|restart|{start,stop,restart}-apachel|
