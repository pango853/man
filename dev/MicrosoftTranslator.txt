en	English
ja	Japanese




echo "#[DEFAULT:ja]" && echo && echo |cat - messages_ja.properties > messages_i18n.properties

python proptranslate.py add --lang=en messages_en.properties
python proptranslate.py add --lang=ja messages_ja.properties
python proptranslate.py translate --lang=en
python proptranslate.py translate --lang=zh-CHT
python proptranslate.py translate --lang=zh-CHS
python proptranslate.py build
